Add to `pkg/metrics/metrics.go` (after existing content, before last closing lines):

```go
import (
    // existing imports...
    "sync/atomic"
    "sync"
)

// PoolStat holds counters for a single object pool.
type PoolStat struct {
    Gets  int64  // total Get() calls (items retrieved from pool)
    Puts  int64  // total Put() calls (items returned to pool)
}

// Active returns an estimate of currently borrowed items (Gets - Puts).
// May be negative during warm-up due to initial pool allocation.
func (s PoolStat) Active() int64 {
    return s.Gets - s.Puts
}

var (
    poolMu    sync.RWMutex
    poolStats = map[string]*poolCounter{}
)

type poolCounter struct {
    gets int64
    puts int64
}

func getOrCreateCounter(name string) *poolCounter {
    poolMu.RLock()
    c, ok := poolStats[name]
    poolMu.RUnlock()
    if ok {
        return c
    }
    poolMu.Lock()
    defer poolMu.Unlock()
    if c, ok = poolStats[name]; ok {
        return c
    }
    c = &poolCounter{}
    poolStats[name] = c
    return c
}

// RecordPoolGet records a Get() call for the named pool.
// Call this immediately after retrieving an object from a sync.Pool.
func RecordPoolGet(name string) int64 {
    c := getOrCreateCounter(name)
    return atomic.AddInt64(&c.gets, 1)
}

// RecordPoolPut records a Put() call for the named pool.
// Call this immediately after returning an object to a sync.Pool.
func RecordPoolPut(name string) int64 {
    c := getOrCreateCounter(name)
    return atomic.AddInt64(&c.puts, 1)
}

// GetPoolStats returns a snapshot of all pool counters.
// The map is keyed by pool name. Pre-populates standard pool names with zero values
// so callers can always find "tokenizer", "parser", and "ast" keys.
func GetPoolStats() map[string]PoolStat {
    // Ensure standard pools are always present
    for _, name := range []string{"tokenizer", "parser", "ast"} {
        getOrCreateCounter(name)
    }

    poolMu.RLock()
    defer poolMu.RUnlock()
    result := make(map[string]PoolStat, len(poolStats))
    for name, c := range poolStats {
        result[name] = PoolStat{
            Gets: atomic.LoadInt64(&c.gets),
            Puts: atomic.LoadInt64(&c.puts),
        }
    }
    return result
}

// ResetPoolStats zeroes all pool counters. Primarily for testing.
func ResetPoolStats() {
    poolMu.Lock()
    defer poolMu.Unlock()
    poolStats = map[string]*poolCounter{}
}
```

```go
// cmd/gosqlx/cmd/stats.go
package cmd

import (
	"encoding/json"
	"fmt"
	"sort"

	"github.com/ajitpratap0/GoSQLX/pkg/metrics"
	"github.com/spf13/cobra"
)

var statsCmd = &cobra.Command{
	Use:   "stats",
	Short: "Show pool utilization statistics",
	Long:  "Display current object pool utilization counters (gets, puts, active borrows).",
	RunE: func(cmd *cobra.Command, args []string) error {
		jsonOutput, _ := cmd.Flags().GetBool("json")
		stats := metrics.GetPoolStats()

		if jsonOutput {
			b, err := json.MarshalIndent(stats, "", "  ")
			if err != nil {
				return err
			}
			fmt.Println(string(b))
			return nil
		}

		// Human-readable table
		fmt.Printf("%-20s %10s %10s %10s\n", "POOL", "GETS", "PUTS", "ACTIVE")
		fmt.Printf("%-20s %10s %10s %10s\n", "----", "----", "----", "------")

		names := make([]string, 0, len(stats))
		for k := range stats {
			names = append(names, k)
		}
		sort.Strings(names)

		for _, name := range names {
			s := stats[name]
			fmt.Printf("%-20s %10d %10d %10d\n", name, s.Gets, s.Puts, s.Active())
		}
		return nil
	},
}

func init() {
	statsCmd.Flags().Bool("json", false, "Output as JSON")
	rootCmd.AddCommand(statsCmd)
}
```

If `->` and `->>` aren't handled, add them to the binary operator parsing section. In most recursive descent parsers, this is in a method like `parseAdditiveExpr` or `parseComparisonExpr`:

```go
// In the section that parses binary operators, add:
case token.Type == models.TOperator && token.Value == "->":
    return parseBinaryOp("->", left, p.parseUnaryExpr)
case token.Type == models.TOperator && token.Value == "->>":
    return parseBinaryOp("->>", left, p.parseUnaryExpr)
```

The exact implementation depends on the parser's operator precedence table. `->` in PostgreSQL has lower precedence than arithmetic but higher than comparisons. Place it in the additive/multiplicative tier.
